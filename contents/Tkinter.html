    MAKING ELEMNTS: (To make visible anything you make, Use pack() function. It aligns the element.) 
from tkinter import *


#Labels
label = Label(text="This is old text")
label.config(text="This is new text")
label.pack()


#Buttons
def action():
    print("Do something")

#calls action() when pressed
button = Button(text="Click Me", command=action)
button.pack()

#Entries
entry = Entry(width=30)
#Add some text to begin with or Delete text from it.
entry.insert(END, string="Some text to begin with.")
entry.delete(0, END)
#Gets text in entry
print(entry.get())
entry.pack()

#Text
text = Text(height=5, width=30)
#Puts cursor in textbox.
text.focus()
#Adds some text to begin with.
text.insert(END, "Example of multi-line text entry.")
#Get's current value in textbox at line 1, character 0
print(text.get("1.0", END))
text.pack()


#Spinbox
def spinbox_used():
    #gets the current value in spinbox.
    print(spinbox.get())
spinbox = Spinbox(from_=0, to=10, width=5, command=spinbox_used)
spinbox.pack()

#Scale
#Called with current scale value.
def scale_used(value):
    print(value)
scale = Scale(from_=0, to=100, orient=HORIZONTAL, command=scale_used)
scale.set(20)      # Assign a default value.
scale.pack()

#Checkbutton
def checkbutton_used():
    #Prints 1 if On button checked, otherwise 0.
    print(checked_state.get())
#variable to hold on to checked state, 0 is off, 1 is on.
checked_state = IntVar()
checkbutton = Checkbutton(text="Is On?", variable=checked_state, command=checkbutton_used)
checked_state.get()
checkbutton.pack()


#Radiobutton
def radio_used():
    print(radio_state.get())
#Variable to hold on to which radio button value is checked.
radio_state = IntVar()
radiobutton1 = Radiobutton(text="Option1", value=1, variable=radio_state, command=radio_used)
radiobutton2 = Radiobutton(text="Option2", value=2, variable=radio_state, command=radio_used)
radiobutton1.pack()
radiobutton2.pack()


#Listbox
def listbox_used(event):
    # Gets current selection from listbox
    print(listbox.get(listbox.curselection()))

listbox = Listbox(height=4)
fruits = ["Apple", "Pear", "Orange", "Banana"]
for item in fruits:
    listbox.insert(fruits.index(item), item)
listbox.bind("<<ListboxSelect>>", listbox_used)
listbox.pack()

    Layout Management And Alignments:

    

""" There are 3 ways to manage layout in tkinter:
     1. pack(): packs everything together with little personal customization.
     2. place(): uses the given coordinate to puts an item.
     3. grid(): makes a grid system.(usually the best)(not compatible with pack()) """
# 1.
label = Label(text="This is text")
label.pack(side="left")  # default: side="center"


# 2.
label = Label(text="This is text")
label.place(x=100, y=150)


# 3.
label = Label(text="This is text")
label.grid(row=1, column=2)

entry = Entry(width=30)
entry.grid(row=1, column=1, columnspan=2)    # Start from 0.


# You can specify padding for each
label.config(pady=10, padx=5)


    
Using Canvas Widget:

    

""" Canvas is used to put image or different shapes in the GUI and can also support text overly. """

# Here is how to add an image and a text on it.
canvas = Canvas(width=200, height=224, bg="yellow", highlightthickness=0)#"highlightthickness" specifies the outline width.

tomato_img = PhotoImage(file="tomato.png")
canvas.create_image(100, 112, image=tomato_img)    # two first args are coordination of element inside canvas.
timer_text = canvas.create_text(100, 130, text="00:00", fill="white", font=(FONT_NAME, 35, "bold"))

# Change the item using "itemconfig"
canvas.itemconfig(timer_text, text="1")
canvas.pack()
