Seaborn

Seaborn is a Python data visualization library based on Matplotlib. It provides a high-level interface for drawing attractive and informative statistical graphics.  (The Gallery of charts)

It is based on Metaplotlib so you can also control its graphs with Matplotlib functions and attributes.

import seaborn as sns
import matplotlib.pyplot as plt

PLOTS:

# Here is how to make a simple scatter plot with seaborn.

plt.figure(figsize=(8, 4), dpi=200)               # Configuration from matplotlib.

sns.set_theme(style="darkgrid", palette=None)     # setting the color style : {darkgrid, whitegrid, dark, white, ticks}

ax = sns.scatterplot(data=df,
                     x='Release_Date',
                     y='USD_Production_Budget',
                     hue='USD_Worldwide_Gross',   # colour
                     size='USD_Worldwide_Gross',) # dot size

ax.set(ylim=(0, 400000000),
       xlim=(df.Release_Date.min(), df.Release_Date.max()),
       ylabel='Budget in $100 millions',
       xlabel='Year',)

plt.show()

Chart Examples:


# Regplot
# Plot data and a linear regression model fit.
# You can find better types of these plots in the gallery.
ax = sns.regplot(data=df,
                 x='Production_Budget',
                 y='Worldwide_Gross',
                 color='#2f4b7c',
                 scatter=True,                     # Adds a scatter plot underlying the regression plot.
                 scatter_kws = {'alpha': 0.3},     # "scatter_kws" is Additional keyword arguments for scatter plot.
                 line_kws ={"color":'#ff7c43', })  # "line_kws" is Additional keyword arguments for regression line.

# Histogram
sns.histplot(data=df,
             x="age",               # It can only have "x"
             hue="category",        # you can also include "hue" key arg. if you have multiple categories of your "x".
             bins=30)               # "bin" specifies the number of bins in histogram. In this case,
                                    # the data will be divided into 30 equal-width bins.

# lmplot
# Plots multiple regression model fits overlapping each other or in a table order.
sns.lmplot(data=df,
            x='year',
            y='age',
            row = 'category',       # Makes multiple charts based on the number of "category" in multiple rows.
            col = 'category',       # Can be used instead of the upper line.
            hue='category',         # Can be used instead of two upper lines if you want all your lines to be in one graph.
            lowess=True,            # Uses moving-average(rolling-average) to draw the line.()
            aspect=2,               # Aspect-ratio
            scatter_kws = {'alpha': 0.6},
            line_kws = {'color': 'black'},)
